# ===================================
# HOW TO GET FIREBASE CONFIG:
# ===================================
# 
# 1. Go to: https://console.firebase.google.com
# 2. Click "Add project" or select existing project
# 3. Enter project name (e.g., "PlacementOS")
# 4. Click through the setup (you can disable Google Analytics)
# 5. Once created, click the web icon (</>) to add a web app
# 6. Register your app with a nickname (e.g., "PlacementOS Web")
# 7. You'll see a firebaseConfig object like this:
#
#    const firebaseConfig = {
#      apiKey: "AIzaSy...",
#      authDomain: "your-project.firebaseapp.com",
#      projectId: "your-project-id",
#      storageBucket: "your-project.appspot.com",
#      messagingSenderId: "123456789",
#      appId: "1:123456789:web:abc123"
#    };
#
# 8. Copy each value and paste it above
#
# 9. Enable Authentication:
#    - In Firebase Console, go to "Authentication"
#    - Click "Get Started"
#    - Click "Sign-in method" tab
#    - Enable "Email/Password"
#    - Click Save
#
# 10. Enable Firestore (optional for now):
#     - In Firebase Console, go to "Firestore Database"
#     - Click "Create database"
#     - Start in "Test mode"
#     - Choose a location
#
